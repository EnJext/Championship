----НАЛАШТУВАННЯ git

www.gitignore.io -сайт де можна згенерувати текст файлу .gitignore

git config --global user.name "git name"

github
	

----ЛОКАЛЬНИЙ git РЕПОЗИТОРІЙ

git clone - клонування існуючого віддаленого .git репозиторія в поточну папку
з клонуванням всіх файлів віддаленого репозиторія і звязування його з віддаленими репо

git init - створення локал .git репозиторія в поточній папці 



---- ВІДДАЛЕНИЙ git РЕПОЗИТОРІЙ

git remote url name - підключається до віддаленого репо з вказаним url та псевдонімом name

git remote show - показує список доступних віддалнених репо 

git remote add origin - додає віддалений репо origin 

git remote show <назва віддаленого репо> - показує інфу по віддаленому репо

git push - додає зміни локал репо в віддалений репо 

git push -u <назва віддаленого репо> master - використовуєтьяс для першого push 
привязує локал гілку master до гілки master віддаленого репо

git pull додає зміни з віддаленого репо в локальний 




----керування файлами в .git репозиторії 
 
git status - показує статус файлів 
статуси :
- файл не додано 
- файл змінено з останнього коміту
- файл змінено з останнього коміту і зміни збережено
- файли видалені з .git репозиторія
- не показує файли які не змінились після останнього коміта 

git add - додає файли для відслідковування або якщо файл був змінений з останнього
коміта додає змінни до файлу 

git reset <назва файлу/директорії> - відміняє останню дію над файлом

git rm --cached <назва> - перестає відслідковувати файл 

git rm -r --cached <назва папки> - перестане відслідковувати всю папку 

git add .
git reset .
git rm -r --cached . робить вказану дію для всіх файлів в дерикторії



----КОМІТИ 

git commit -m "massege" -робить знімок(commit) в поточній гілці 

commit - змінна між останнім комітом 

в знімок входять таків файли :
- файли що не змінювались з останнього коміта 
- фали які додано за допомогою git add
- файли які були змінені з останнього коміта і зміни було додано з git add
- файли що не знаходяться в .gitignore

git log - видає історію комітів 

git checkout <id коміта> - відкатує до певного коміта з id

----ГІЛКИ

git branch - показує доступні гілки та поточну гілку 

git branch <назва гілки> -створює нову гілку першим комітом якої є 
останій коміт поточної гілки 

git checkout <назва гілки> - переходить у вказану гілку 

git merge <гілка1> <гілка2> зливає гілку1 в гілку2 та свторює коміт в гілці
2 який зберігає всі зміни внаслідок злиття

git branch --merged - покаже уже злиті гілки

git branch -d <назва> -видаляє гілку після злиттся (-D видаляє не злиті гілки)

----ВІДКАТ Коммітів 
git reset --hard HEAD~1 (К-сть комітів на скільк потрібно відкатити) або  hash
работает только для локальной ветки, если коммит запушен то нужно использовать отменяющий коммит

git push -f origin <branchName> - примусовий комітт в remote 

git pull для оновлення 


git revert HEAD - отменяем коммит по указателю HEAD

git commit -m "reverted the last commit" - фиксирует отмену коммита в новом коммите
